def columnInput():
    '''
        purpose: Take column input from the user and then validate that input
        outcome: Valid column name
    '''

    # Take input from the user for the column name
    column = input("Enter the column name: ")

    # Check if the input is a valid column name in the DataFrame
    if column in df.columns:
        # Check if the column contains numerical values
        if df[column].dtype in ['int64', 'float64']:
            return column # if column is valid and contains numerical values
        else:
            # if column doesn't contains numerical values
            print("Column contains non-numeric values. Please enter a column with numeric values") 
            return -1
    else:
        # if input for the column name is invalid
        print("Please enter a valid column name")
        return -1

def sumOfColumn(column):
    '''
    purpose: Calculate the sum of the provided column.
    input: column - list containing the values from particular column
    output: Sum of the column
    '''

    # Initialize the sum to 0
    sum_ = 0

    # Iterate through each value in the column and add it to the sum
    for i in column:
        sum_ += i
    
    # Return the sum of the column
    return sum_

def meanOfColumn(column):
    '''
    purpose: Calculate the mean of the given column.
    input: column - list containing the values from particular column
    output: Mean of the column
    '''

    # Calculate the sum of the column
    sum_ = sumOfColumn(column)

    # Calculate the total number of values in the column
    totalNumber = len(column)

    # Calculate the mean of the column
    mean = sum_ / totalNumber

    # Return the mean of the column
    return mean

def standardDeviation(column):
    '''
    purpose: Calculate the standard deviation of the given column.
    input: column - list containing the values from particular column
    output: Standard deviation of the column
    '''

    # Initialize a variable that stores the value of numerator portion of the standard deviation formula
    numeratorPortion = 0

    # Calculate the total number of values in the column
    totalLines = len(column) - 1 
    
    # Iterate through each value in the column to calculate the numerator portion
    for i in column: 
        numeratorPortion += ((i - meanOfColumn(column)) ** 2)
    
    # Calculate the standard deviation
    standardDeviation = (numeratorPortion / totalLines) ** 0.5
    
    # Return the standard deviation of the column
    return standardDeviation

def skewness(column):
    '''
    purpose: Calculate the skewness of the given column.
    input: column - list containing the values from particular column
    output: Skewness of the column
    '''

    # Initialize a variable that stores the value of numerator portion of the skewness formula
    numeratorPortion = 0
    
    # Calculate the total number of values in the column
    totalLines = len(column)
    
    # Iterate through each value in the column to calculate the numerator portion
    for i in column:
        numeratorPortion += ((i - meanOfColumn(column)) ** 3)
    
    # Calculate the value of denominator portion of the skewness formula
    denominatorPortion = (totalLines - 1) * (standardDeviation(column) ** 3)
    
    # Calculate the skewness 
    skewness = numeratorPortion / denominatorPortion
    
    # Return the skewness of the column
    return skewness

def kurtosis(column):
    '''
    purpose: Calculate the kurtosis of the given column.
    input: column - list containing the values from particular column
    output: Kurtosis of the column
    '''

    # Initialize a variable that stores the value of numerator portion of the kurtosis formula
    numeratorPortion = 0
    
    # Calculate the total number of values in the column
    totalLines = len(column)
    
    # Iterate through each value in the column to calculate the numerator portion
    for i in column:
        numeratorPortion += ((i - meanOfColumn(column)) ** 4)
    
    # Calculate the denominator portion of the kurtosis formula
    denominatorPortion = (totalLines - 1) * (standardDeviation(column) ** 4)
    
    # Calculate the kurtosis
    kurtosis = (numeratorPortion / denominatorPortion) - 3
    
    # Return the kurtosis of the column
    return kurtosis

def correlation(column1, column2):
    '''
    purpose: Calculate the correlation coefficient between two columns.
    input:
        column1 - list containing the values from first column
        column2 - list containing the values from second column
    output: Correlation coefficient between the two columns
    '''

    # Calculate the mean of each column
    mean1 = meanOfColumn(column1)
    mean2 = meanOfColumn(column2)

    # Initialize variables for the numerator and denominators of the correlation formula
    numerator = 0
    denominator1 = 0
    denominator2 = 0
    
    # Iterate through the values of both columns to calculate the numerator and denominators
    for i in range(len(column1)):
        numerator += (column1[i] - mean1) * (column2[i] - mean2)
        denominator1 += (column1[i] - mean1) ** 2
        denominator2 += (column2[i] - mean2) ** 2

    # Calculate the correlation coefficient 
    correlation = numerator / ((denominator1 * denominator2) ** 0.5)
    
    # Return the correlation coefficient
    return correlation
